Code 
Freshmen <- passing|>
  filter(Year == "FR")

passing_ratio <- passing |>
  group_by(Year) |>
  summarise(
    Interceptions = mean(Int),
    Touchdowns = mean(TD),
  )
  
passing_av <- passing |>
  group_by(Year) |>
  summarise(
    Completions = mean(Comp),
    Attempts = mean(Att),
  )|>
  pivot_longer(
    cols = -Year,
    names_to = "Stat",
    values_to = "Total"
  )
