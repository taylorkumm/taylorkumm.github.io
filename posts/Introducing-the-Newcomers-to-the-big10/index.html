firstdownoff <- read_csv("firstdownoff.csv")
teamstats23 <- read_csv("~/Desktop/Sports Data V&A SPMC350/Blog/posts/Introducing-the-Newcomers-to-the-big10/cfbstats/cfb23.csv") |>
  separate(Team, into=c("School", "Conference"), sep="\\(") |>
  mutate(Conference=gsub(")", "", Conference))

oregon <- teamstats23 |>
  filter(School == "Oregon ")
ucla <- teamstats23 |>
  filter(School == "UCLA ")
usc <- teamstats23 |>
  filter(School == "Southern California ")
wash <- teamstats23 |>
  filter(School == "Washington ")
bigten23 <- teamstats23|>
  filter(Conference == "Big Ten")
futurebig <- teamstats23|>
  filter(`Big Ten in 2024`=="YES")

playsstacked <- firstdownoff |> 
  select(TEAM, `RUFD`, `PAFD`) |> 
  pivot_longer(
    cols = -TEAM,
    names_to = "Type",
    values_to = "Plays"
  )

ggplot() + 
  geom_bar(data=playsstacked, aes(x=reorder(TEAM,Plays), weight=Plays, fill=Type)) +
  scale_fill_manual(values = c("skyblue","forestgreen"))+
